class Cat extends Animal {
  constructor(name, age, status) {
    super(name, age, 4, 'cat', status);
  }

  introduce() {

  }

Cubes and Setters 

class SiegeState {
  constructor(canMove, damage) {
    this.canMove = false;
    this.damage = 20;  
  }
}

class TankState {
  constructor(canMove, damage) {
    this.canMove = true;
    this.damage = 5;  
  }
}

class Tank {
  constructor(state, canMove, damage) {
    this.state = new TankState();    
  }

  get canMove() {
    return this.state.canMove;
  }
  get damage() {
    return this.state.damage;
  }
}
